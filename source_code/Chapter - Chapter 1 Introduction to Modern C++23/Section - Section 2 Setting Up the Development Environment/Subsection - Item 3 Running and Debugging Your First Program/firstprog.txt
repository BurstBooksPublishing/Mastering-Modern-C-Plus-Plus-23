#include <cstddef>
#include <cstdlib>
#include <iostream>
#include <stdexcept>
#include <vector>

int main(int argc, char* argv[]) {
    if (argc != 2) {
        std::cerr << "usage: " << argv[0] << " <positive-integer>\n";
        return EXIT_FAILURE;
    }

    try {
        const std::size_t n = std::stoull(argv[1]);
        if (n == 0) {
            std::cerr << "error: n must be positive\n";
            return EXIT_FAILURE;
        }

        std::vector<long long> data(n, 1LL);
        long long sum = 0;

        for (std::size_t i = 0; i < n; ++i) {
            sum += data[i];
        }

        std::cout << sum << '\n';
    } catch (const std::invalid_argument&) {
        std::cerr << "error: invalid number\n";
        return EXIT_FAILURE;
    } catch (const std::out_of_range&) {
        std::cerr << "error: number out of range\n";
        return EXIT_FAILURE;
    }

    return EXIT_SUCCESS;
}
