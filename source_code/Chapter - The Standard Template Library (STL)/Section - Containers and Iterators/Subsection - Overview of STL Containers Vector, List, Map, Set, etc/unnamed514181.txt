#include <iostream>
#include <vector>
#include <list>
#include <map>
#include <set>

int main() {
    std::vector<int> vec{1, 2, 3, 4, 5};
    vec.push_back(6);
    std::cout << "Vector elements: ";
    for (const int v : vec) std::cout << v << ' ';
    std::cout << '\n';

    std::list<int> lst{10, 20, 30};
    lst.push_front(5);
    std::cout << "List elements: ";
    for (const int l : lst) std::cout << l << ' ';
    std::cout << '\n';

    std::map<std::string, int> mp{{"apple", 1}, {"banana", 2}};
    mp["cherry"] = 3;
    std::cout << "Map elements: ";
    for (const auto& [key, value] : mp) std::cout << key << ':' << value << ' ';
    std::cout << '\n';

    std::set<int> st{5, 3, 1, 4, 2};
    st.insert(6);
    std::cout << "Set elements: ";
    for (const int s : st) std::cout << s << ' ';
    std::cout << '\n';
}
