#include <algorithm>
#include <iostream>
#include <iterator>
#include <ranges>
#include <vector>

int main() {
    const std::vector<int> v{1, 2, 3, 4, 5, 6, 7, 8, 9};

    auto even_squares = v | std::views::filter([](int n) { return n % 2 == 0; })
                          | std::views::transform([](int n) { return n * n; });

    std::ranges::copy(even_squares, std::ostream_iterator<int>(std::cout, " "));
    std::cout << '\n';
}
