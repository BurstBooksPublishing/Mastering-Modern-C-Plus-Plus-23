#include <algorithm>
#include <iostream>
#include <vector>

int main() {
    std::vector<int> numbers{4, 2, 3, 1, 5};

    std::sort(numbers.begin(), numbers.end());  // ascending by default

    for (int num : numbers) {
        std::cout << num << ' ';
    }
    std::cout << '\n';
    return 0;
}
