#include <filesystem>
#include <fstream>
#include <iostream>
#include <string>

void read_file(const std::filesystem::path& file_path) {
    if (!std::filesystem::exists(file_path)) {
        std::cerr << "File not found: " << file_path << '\n';
        return;
    }

    std::ifstream file(file_path, std::ios::in);
    if (!file) {
        std::cerr << "Cannot open file: " << file_path << '\n';
        return;
    }

    std::string line;
    while (std::getline(file, line)) {
        std::cout << line << '\n';
    }
}
