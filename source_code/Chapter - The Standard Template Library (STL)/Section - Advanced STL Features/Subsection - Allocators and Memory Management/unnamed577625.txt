#include <memory>
#include <vector>
#include <iostream>

int main() {
    std::allocator<int> alloc;

    int* arr = alloc.allocate(5);

    std::uninitialized_fill_n(arr, 5, 1);   // fill with 1
    for (std::size_t i = 0; i < 5; ++i) arr[i] += i;  // 1..5

    for (std::size_t i = 0; i < 5; ++i) std::cout << arr[i] << ' ';
    std::cout << '\n';

    std::destroy_n(arr, 5);
    alloc.deallocate(arr, 5);

    std::vector<int, std::allocator<int>> vec{10, 20};
    for (int v : vec) std::cout << v << ' ';
    std::cout << '\n';
}
