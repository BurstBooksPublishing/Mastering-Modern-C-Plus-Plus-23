#include <iostream>
#include <stdexcept>

void check_age(int age) {
    if (age < 0) {
        throw std::invalid_argument("Age cannot be negative");
    }
    std::cout << "Age is valid: " << age << '\n';
}

int main() {
    try {
        check_age(-5);
    } catch (const std::invalid_argument& e) {
        std::cerr << "Error: " << e.what() << '\n';
    }
    return 0;
}
