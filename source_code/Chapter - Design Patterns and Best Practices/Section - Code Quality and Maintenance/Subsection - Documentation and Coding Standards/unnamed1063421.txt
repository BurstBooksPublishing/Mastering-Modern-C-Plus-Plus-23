#!/usr/bin/env bash
# Safe, production-ready wrapper for committing the fix

set -euo pipefail

# Ensure we are inside a Git repository
git rev-parse --is-inside-work-tree >/dev/null || {
  echo "Error: not in a Git repository" >&2
  exit 1
}

# Stage only the file(s) that contain the fix for division-by-zero
git add --update -- ':(glob)**/*compute*' ':(glob)**/*value*'

# Commit with the provided message
git commit -m "Fix division by zero in computeValue()"
