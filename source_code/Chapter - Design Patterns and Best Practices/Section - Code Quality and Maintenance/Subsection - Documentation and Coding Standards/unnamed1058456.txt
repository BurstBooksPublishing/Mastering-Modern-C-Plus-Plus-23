#include <stdexcept>
#include <limits>

// Returns n! for 0 ≤ n ≤ 12 (fits in 32-bit int).
// Throws std::invalid_argument for out-of-range input.
constexpr int factorial(int n) {
    if (n < 0 || n > 12)
        throw std::invalid_argument("factorial: input out of range");

    int result = 1;
    for (int i = 2; i <= n; ++i)
        result *= i;
    return result;
}
