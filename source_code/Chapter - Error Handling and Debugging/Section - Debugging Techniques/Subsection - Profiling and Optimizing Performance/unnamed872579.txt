#include <iostream>
#include <limits>

// Returns the sum of the first n natural numbers (1..n).
// Undefined behaviour if the result overflows int.
int sum_of_natural_numbers(int n) {
    return n * (n + 1) / 2;
}

int main() {
    constexpr int n = 10'000;          // Safe for 32-bit int
    const int sum = sum_of_natural_numbers(n);
    std::cout << "Sum of first " << n << " natural numbers: " << sum << '\n';
    return 0;
}
