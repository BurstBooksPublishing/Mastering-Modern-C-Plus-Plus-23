#include <iostream>
#include <vector>
#include <span>
#include <algorithm>

int main() {
    const std::vector<int> values{10, 20, 30, 40, 50, 60};

    // Safely extract the middle two elements (indices 2 and 3)
    const size_t start = 2;
    const size_t count = 2;
    if (start + count <= values.size()) {
        std::span<const int> mid{values.data() + start, count};
        std::ranges::copy(mid, std::ostream_iterator<int>(std::cout, " "));
    }

    return 0;
}
