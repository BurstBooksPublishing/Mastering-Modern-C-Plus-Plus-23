#include <vector>
#include <algorithm>
#include <execution>

int main() {
    std::vector<int> data{1, 2, 3, 4, 5};

    // Double every element in parallel
    std::for_each(std::execution::par_unseq,
                  data.begin(), data.end(),
                  [](int& item) { item *= 2; });

    return 0;
}
