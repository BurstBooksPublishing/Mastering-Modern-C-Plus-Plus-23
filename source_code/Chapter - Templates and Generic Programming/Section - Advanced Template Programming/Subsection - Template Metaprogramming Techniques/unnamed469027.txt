#include <type_traits>
#include <iostream>

template <typename T>
void process(const T& value) {
    if constexpr (std::is_integral_v<T>) {
        std::cout << "Integral: " << value << '\n';
    } else if constexpr (std::is_floating_point_v<T>) {
        std::cout << "Floating: " << value << '\n';
    } else {
        std::cout << "Other: " << value << '\n';
    }
}

int main() {
    process(42);
    process(3.14);
}
