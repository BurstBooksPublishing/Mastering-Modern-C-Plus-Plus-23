#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
#include <ranges>
#include <format>

// Example: minimal, production-ready C++23 program
int main() {
    const std::vector<int> data{3, 1, 4, 1, 5, 9};

    // Transform and collect even numbers, squared
    auto evens = data
               | std::views::filter([](int n){ return n % 2 == 0; })
               | std::views::transform([](int n){ return n * n; })
               | std::ranges::to<std::vector>();

    // Output results
    for (int v : evens) std::cout << std::format("{}\n", v);
}
