#include <iostream>
#include <format>
#include <ranges>
#include <vector>
#include <algorithm>

int main() {
    std::cout << std::format("Welcome to Modern C++23!\n");

    auto numbers = std::views::iota(1, 10)
                 | std::views::filter([](int n) { return n % 2 == 0; })
                 | std::views::transform([](int n) { return n * 2; });

    for (int n : numbers) std::cout << n << ' ';
    std::cout << '\n';
}
