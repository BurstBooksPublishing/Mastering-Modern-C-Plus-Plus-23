#include <iostream>
#include <thread>
#include <stdexcept>

void threadFunction() {
    std::cout << "Hello from thread!" << std::endl;
}

int main() {
    std::thread t(threadFunction);

    try {
        if (t.joinable()) {
            t.join();
        }
    } catch (const std::system_error& e) {
        std::cerr << "Thread join failed: " << e.what() << '\n';
        return EXIT_FAILURE;
    }

    return EXIT_SUCCESS;
}
