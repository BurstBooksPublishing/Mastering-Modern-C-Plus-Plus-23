#include <algorithm>
#include <iostream>
#include <vector>

// Erase-remove idiom: single-pass, O(n) complexity, preserves relative order.
void remove_evens(std::vector<int>& v) {
    v.erase(std::remove_if(v.begin(), v.end(),
                           [](int x) { return x % 2 == 0; }),
            v.end());
}

int main() {
    std::vector<int> a{1, 2, 3, 4, 5, 6};
    remove_evens(a);
    for (int x : a) std::cout << x << ' '; // 1 3 5
    std::cout << '\n';
}
