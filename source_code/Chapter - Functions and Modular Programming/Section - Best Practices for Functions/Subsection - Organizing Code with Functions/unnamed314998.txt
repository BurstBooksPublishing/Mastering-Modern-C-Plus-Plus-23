#include <algorithm>
#include <iostream>
#include <numeric>
#include <vector>

int calculateSum(const std::vector<int>& numbers) {
    return std::accumulate(numbers.begin(), numbers.end(), 0);
}

int findMax(const std::vector<int>& numbers) {
    if (numbers.empty()) throw std::invalid_argument("Empty vector");
    return *std::max_element(numbers.begin(), numbers.end());
}

void displayResults(int sum, int max) {
    std::cout << "Sum of elements: " << sum << '\n'
              << "Maximum element: " << max << '\n';
}

int main() {
    const std::vector<int> numbers{10, 20, 30, 40, 50};

    const int sum = calculateSum(numbers);
    const int max = findMax(numbers);
    displayResults(sum, max);
}
