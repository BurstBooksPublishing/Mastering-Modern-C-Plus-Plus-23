#include <algorithm>
#include <iostream>
#include <iterator>
#include <ranges>
#include <vector>

int main() {
    const std::vector<int> numbers{3, 1, 4, 1, 5, 9, 2, 6};

    // Stream even numbers directly to stdout
    std::ranges::copy(
        numbers | std::views::filter([](int n) { return n % 2 == 0; }),
        std::ostream_iterator<int>(std::cout, " ")
    );
}
