#include <iostream>
#include <stdexcept>

int main() {
    try {
        throw std::out_of_range("Index out of range");
    } catch (const std::out_of_range& e) {
        std::cerr << "Caught out_of_range: " << e.what() << '\n';
    } catch (const std::exception& e) {
        std::cerr << "Caught unexpected exception: " << e.what() << '\n';
    }
    return 0;
}
