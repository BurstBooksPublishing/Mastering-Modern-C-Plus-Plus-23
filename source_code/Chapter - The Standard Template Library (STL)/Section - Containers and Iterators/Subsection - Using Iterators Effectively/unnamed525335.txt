#include <algorithm>
#include <iostream>
#include <vector>

int main() {
    std::vector<int> numbers{1, 2, 3, 4, 5};

    // Double every element in place
    std::for_each(numbers.begin(), numbers.end(), [](int& n) { n *= 2; });

    // Output the modified vector once
    for (const int n : numbers) std::cout << n << ' ';
    std::cout << '\n';

    return 0;
}
