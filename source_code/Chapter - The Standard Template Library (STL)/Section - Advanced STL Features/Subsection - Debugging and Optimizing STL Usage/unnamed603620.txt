#include <vector>
#include <ranges>
#include <iostream>

int main() {
    std::vector<int> vec{1, 2, 3, 4, 5};

    // Create a lazy view of even numbers
    auto even = vec | std::views::filter([](int x) { return x % 2 == 0; });

    // Materialize the view into a new container for safe reuse
    std::vector<int> evens{even.begin(), even.end()};
}
