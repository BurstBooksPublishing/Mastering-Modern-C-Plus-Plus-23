#include <algorithm>
#include <cassert>
#include <cstdint>
#include <iostream>
#include <numeric>
#include <span>
#include <vector>

int calculateSum(std::span<const int> numbers) {
    return std::accumulate(numbers.begin(), numbers.end(), 0);
}

void demonstrateBestPractices() {
    const std::vector<int> data{1, 2, 3, 4, 5};

    const int sum = calculateSum(data);
    std::cout << "Sum of elements: " << sum << '\n';

    assert(!data.empty() && "Data should not be empty!");
}

int main() {
    demonstrateBestPractices();
    return 0;
}
