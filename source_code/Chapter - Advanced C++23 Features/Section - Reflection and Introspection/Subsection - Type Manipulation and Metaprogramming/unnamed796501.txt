#include <iostream>
#include <type_traits>
#include <concepts>

// Restrict to arithmetic types only (C++20 concept)
template<std::integral T>
constexpr T add(T a, T b) noexcept {
    return a + b;
}

template<std::floating_point T>
constexpr T add(T a, T b) noexcept {
    return a + b;
}

int main() {
    std::cout << add(10, 20) << '\n';          // 30
    // std::cout << add("hello", "world") << '\n'; // compile-time error
}
