#include <fstream>
#include <iostream>
#include <string>

int main() {
    const std::string file_path = "example.txt";
    std::ifstream input_file(file_path, std::ios::in);

    if (!input_file.is_open()) {
        std::cerr << "Error: Failed to open file '" << file_path << "'.\n";
        return 1;
    }

    int value;
    while (input_file >> value) {
        // Process value
    }

    if (input_file.fail() && !input_file.eof()) {
        std::cerr << "Error: Invalid data encountered while reading '" << file_path << "'.\n";
        return 1;
    }

    return 0;
}
