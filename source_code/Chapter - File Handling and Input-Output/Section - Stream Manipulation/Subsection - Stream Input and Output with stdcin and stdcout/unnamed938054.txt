#include <iostream>
#include <iomanip>
#include <limits>

int main() {
    int age{};
    double salary{};

    std::cout << "Enter your age: ";
    while (!(std::cin >> age) || age < 0) {
        std::cin.clear();
        std::cin.ignore(std::numeric_limits<std::streamsize>::max(), '\n');
        std::cout << "Invalid age. Please enter a non-negative integer: ";
    }

    std::cout << "Enter your salary: ";
    while (!(std::cin >> salary) || salary < 0.0) {
        std::cin.clear();
        std::cin.ignore(std::numeric_limits<std::streamsize>::max(), '\n');
        std::cout << "Invalid salary. Please enter a non-negative number: ";
    }

    std::cout << "\nAge:    " << std::setw(10) << age << '\n'
              << "Salary: $" << std::fixed << std::setprecision(2) << salary << '\n';
    return 0;
}
