#include <algorithm>
#include <iostream>
#include <vector>
#include <ranges>

int main() {
    std::vector<int> numbers{3, 1, 4, 1, 5, 9};

    std::ranges::sort(numbers);          // ascending order

    for (const int n : numbers)          // print result
        std::cout << n << ' ';
    std::cout << '\n';

    return 0;
}
