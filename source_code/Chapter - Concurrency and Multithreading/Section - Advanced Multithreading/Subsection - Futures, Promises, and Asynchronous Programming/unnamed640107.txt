#include <iostream>
#include <future>
#include <thread>

void compute_value(std::promise<int> prom) {
    prom.set_value(42);
}

int main() {
    std::promise<int> prom;
    std::future<int> fut = prom.get_future();

    std::thread t(compute_value, std::move(prom));
    t.join();

    std::cout << "The value is: " << fut.get() << '\n';
    return 0;
}
