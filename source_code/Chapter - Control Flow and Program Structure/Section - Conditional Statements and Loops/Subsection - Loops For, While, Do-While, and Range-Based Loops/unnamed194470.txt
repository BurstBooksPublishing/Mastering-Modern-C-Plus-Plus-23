#include <iostream>
#include <vector>
#include <stdexcept>

int main() {
    try {
        // For loop
        std::cout << "For loop:\n";
        for (int i = 0; i < 5; ++i) {
            std::cout << "i = " << i << '\n';
        }

        // While loop
        std::cout << "\nWhile loop:\n";
        int j = 0;
        while (j < 5) {
            std::cout << "j = " << j << '\n';
            ++j;
        }

        // Do-While loop
        std::cout << "\nDo-While loop:\n";
        int k = 0;
        do {
            std::cout << "k = " << k << '\n';
            ++k;
        } while (k < 5);

        // Range-based for loop
        std::cout << "\nRange-based for loop:\n";
        const std::vector<int> numbers{1, 2, 3, 4, 5};
        for (const int num : numbers) {
            std::cout << "num = " << num << '\n';
        }
    } catch (const std::exception& ex) {
        std::cerr << "Error: " << ex.what() << '\n';
        return EXIT_FAILURE;
    }
    return EXIT_SUCCESS;
}
