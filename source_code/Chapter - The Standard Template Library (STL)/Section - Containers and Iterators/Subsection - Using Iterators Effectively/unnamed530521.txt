#include <algorithm>
#include <iostream>
#include <iterator>
#include <vector>
#include <ranges>

int main() {
    std::vector<int> vec{1, 2, 3, 4, 5, 6};

    auto is_even = [](int n) { return n % 2 == 0; };

    std::ranges::copy(
        vec | std::views::filter(is_even),
        std::ostream_iterator<int>(std::cout, " ")
    );
}
