#include <algorithm>
#include <execution>
#include <iostream>
#include <vector>

int main() {
    std::vector<int> vec{1, 2, 3, 4, 5};

    // Double every element in parallel
    std::for_each(std::execution::par_unseq, vec.begin(), vec.end(),
                  [](int& n) { n *= 2; });

    // Output results
    for (const int n : vec) std::cout << n << ' ';
    std::cout << '\n';
}
