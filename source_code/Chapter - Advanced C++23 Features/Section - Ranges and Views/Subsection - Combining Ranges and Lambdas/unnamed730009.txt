#include <iostream>
#include <ranges>
#include <vector>

int main() {
    // Generate first 10 Fibonacci numbers
    auto fib = std::views::iota(0) | std::views::transform([](int n) {
        unsigned long long a = 0, b = 1;
        for (int i = 0; i < n; ++i) {
            auto next = a + b;
            a = b;
            b = next;
        }
        return a;
    }) | std::views::take(10);

    // Output sequence
    for (auto v : fib) std::cout << v << ' ';
    std::cout << '\n';
    return 0;
}
