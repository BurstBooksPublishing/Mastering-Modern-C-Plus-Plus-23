#include <algorithm>
#include <iostream>
#include <vector>

int main() {
    std::vector<int> numbers{1, 2, 3, 4, 5};

    // Double every element in-place
    std::for_each(numbers.begin(), numbers.end(), [](int& n) { n *= 2; });

    // Output results
    for (const int num : numbers) std::cout << num << ' ';
    std::cout << '\n';
}
