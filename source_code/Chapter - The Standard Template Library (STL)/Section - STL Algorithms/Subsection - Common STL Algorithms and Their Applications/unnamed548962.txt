#include <algorithm>
#include <iostream>
#include <vector>

int main() {
    std::vector<int> vec{5, 3, 1, 4, 2};

    std::sort(vec.begin(), vec.end());                       // ascending
    std::cout << "Sorted: ";
    for (int v : vec) std::cout << v << ' ';
    std::cout << '\n';

    auto min_it = std::min_element(vec.begin(), vec.end());  // smallest value
    std::cout << "Minimum: " << *min_it << '\n';

    std::reverse(vec.begin(), vec.end());                    // descending
    std::cout << "Reversed: ";
    for (int v : vec) std::cout << v << ' ';
    std::cout << '\n';

    int count = std::count(vec.begin(), vec.end(), 3);       // occurrences of 3
    std::cout << "Count of 3: " << count << '\n';

    bool all_even = std::all_of(vec.begin(), vec.end(),
                                [](int v) { return v % 2 == 0; });
    std::cout << "All even? " << (all_even ? "Yes" : "No") << '\n';
}
