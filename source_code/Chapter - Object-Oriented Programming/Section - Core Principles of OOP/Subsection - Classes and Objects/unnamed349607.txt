#include <stdexcept>

class Box {
public:
    int width  = 1;
    int height = 1;
    int depth  = 1;

    Box() = default;

    Box(int w, int h, int d) {
        if (w <= 0 || h <= 0 || d <= 0)
            throw std::invalid_argument("Dimensions must be positive");
        width  = w;
        height = h;
        depth  = d;
    }
};
