#include <iostream>

int main() {
    const int z = 2;

    switch (z) {
        case 1:
            std::cout << "z is 1\n";
            break;
        case 2:
            std::cout << "z is 2\n";
            break;
        default:
            std::cout << "z is neither 1 nor 2\n";
            break;
    }

    return 0;
}
