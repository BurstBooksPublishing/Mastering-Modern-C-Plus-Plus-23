#include <array>
#include <memory_resource>
#include <vector>

int main() {
    // Static storage for the monotonic resource
    alignas(std::max_align_t) std::array<std::byte, 1024> buffer;

    // Stack-based memory resource; no dynamic allocation
    std::pmr::monotonic_buffer_resource pool{buffer.data(), buffer.size()};

    // Vector using the stack resource
    std::pmr::vector<int> vec{&pool};
    vec.push_back(42);
}
